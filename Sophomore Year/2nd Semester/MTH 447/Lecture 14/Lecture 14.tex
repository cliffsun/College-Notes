\begin{theorem}
    Assume $z_k \in \Lambda$, $z_k \in \mathbb{R}$, $\lim_{k\rightarrow \infty} = z_0$, then $z_0 \in \Lambda$.
\end{theorem}

\begin{definition}
    Let $C \subseteq \mathbb{R}$. We say that $C$ is sequentially closed if whenever $x_n \in C$, $x_n \rightarrow x_0 \implies x_0 \in C$. 
\end{definition}

\begin{proof}
    $z_k \rightarrow z_0$. For all $\epsilon > 0$, $\exists N$ s.t. if $n > N$, 
    \begin{equation}
        |z_0 - z_k| < \epsilon
    \end{equation}
    $\exists \delta > 0$ s.t. 
    \begin{equation}
        (z_k - \delta, z_k + \delta) \subseteq (z_0 - \epsilon, z_0 + \epsilon)
    \end{equation}
\end{proof}

\subsubsection*{4 different characterizations of lim sup}
\begin{enumerate}
    \item $\lim_{n\rightarrow \infty} \sup_{k > n} \{x_k\}$
    \item $M = \lim \sup x_n$ and for any $\epsilon > 0$, $\{n \in \mathbb{N} : |x_n - M| < \epsilon\}$ is infinite. 
    \item Upper Fuzzy Bounds (see homework)
    \item If $M = \lim\sup x_n$, then 
    \begin{enumerate}
        \item $\exists$ subseq $x_{n_k} \rightarrow M$
        \item $\nexists$ subseteq $x_{n_{k}} \rightarrow \hat{M}$ for any $\hat{M} > M$
    \end{enumerate}
\end{enumerate}

\subsubsection*{Fun Facts about lim sup}

\begin{enumerate}
    \item If $x_n \rightarrow L$, then $\lim \sup (x_n + y_n) = L + \lim \sup (y_n)$
    \item If $x_n \rightarrow L$ and $L > 0$, then $\lim \sup (x_n y_n) = L\lim\sup (y_n)$
    \item Let $x_n$ be a sequence. THen $x_n \neq 0$ for all $n$. Then 
    \begin{equation}
        \lim \inf \bigg|\frac{x_{n+1}}{x_n} \bigg| \leq \lim \inf \big|x_n\big|^{\frac{1}{n}} \leq \lim \sup \big| x_n \big| ^{\frac{1}{n}} \leq \lim \sup \bigg| \frac{x_{n+1}}{x_n}\bigg|
    \end{equation}
\end{enumerate}
