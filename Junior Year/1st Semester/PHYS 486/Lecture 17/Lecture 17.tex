\section*{Basic Rules}

\begin{enumerate}
    \item physical state $\ket{\psi}$ in Hilbert Space. Must be normalizable 
    \item Observables with real eigenvalues $\hat{A}\ket{\alpha_n} = a_n\ket{\alpha_n}$
    \item Born's rule: $Pr(a_n) = |\braket{\alpha_n}{\psi}|^2$
    \item Expectation values: $\hat{A} = \braket{\psi}{\hat{A}\psi}$
    \item Stationary states: $\hat{H}\psi = E\psi$
    \item Time evolution of the stationary states: $\psi_n \rightarrow \psi_n\exp(-iE_n/\hbar t)$
\end{enumerate}

\subsection*{Typical Problems (cont. and dis. basis)}

\begin{enumerate}
    \item Validity of states, is it normalized, real eigenvalues of observables
    \item Bases: inner product, write states in eigenbasis of some operator
    \item Basis transformation operator:
    \begin{equation}
        \sum_k \ket{k}\bra{k}
    \end{equation}
    \item Probabilities for measurement outcomes. Statistics for measurement outcome.
    \begin{equation}
        \langle A \rangle, \;\sigma_A^2
    \end{equation}
\end{enumerate}

\section*{Wave mechanics}

Continuous basis $\ket{x}$, $\ket{p}$. (1-dimensional)

\begin{enumerate}
    \item SWE: $H = \frac{p^2}{2m} + V(x) \iff -\frac{\hbar^2}{2m}\frac{\partial^2}{\partial x^2} + \hat{V}(x)$
    \item Inner product: $\braket{\psi}{\varphi} = \int dx \psi^*(x)\varphi(x)$
    \item Actual problem: choose $V(x)$. Need $\psi,\psi'$ continuous. If $V \rightarrow \infty$, then only $\psi$ needs to be continuous. 
\end{enumerate}

\subsection*{Typical Tasks}

\begin{enumerate}
    \item Given some potential, find s.s. and energies. 
    \item Find $\langle x \rangle$, $\langle p \rangle$, $\sigma_x$, $\sigma_p$. Given some $\psi(x)$. 
    \item Probabilities for "range" of outcomes. 
\end{enumerate}

\section*{Specific problems}

\begin{enumerate}
    \item Free Particle, $V=0$. The wave function must be continuous. SS. Plane waves are not physical. BUT, good basis. 
        \item Consequence: No such thing as a completely still standing of a free particle. 
    \item Infinite square well: Only has bound states (states that are trapped in this well) $\ket{k}$ and $E_k \propto k^2$
    \item Finite square well: Both $\psi$ and $\psi'$ need to be continuous.  
\end{enumerate}
